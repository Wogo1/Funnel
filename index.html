    function detectLang() {
      // 1) Query param ?lang=
      const params = new URLSearchParams(window.location.search);
      const q = (params.get('lang') || '').toLowerCase();
      if (q.startsWith('en')) return 'en';
      if (q.startsWith('nl')) return 'nl';

      // 2) <html lang="...">
      const hl = (document.documentElement.lang || '').toLowerCase();
      if (hl.startsWith('en')) return 'en';
      if (hl.startsWith('nl')) return 'nl';

      // 3) path /en /nl (au cas où)
      const p = (window.location.pathname || '').toLowerCase();
      if (p === '/en' || p.startsWith('/en/')) return 'en';
      if (p === '/nl' || p.startsWith('/nl/')) return 'nl';

      // 4) navigator
      const nav = (navigator.language || '').toLowerCase();
      if (nav.startsWith('en')) return 'en';
      if (nav.startsWith('nl')) return 'nl';

      return 'fr';
    }

    function normalizeStep(v) { return (v || '').toString().trim().toLowerCase(); }

    function showStep(stepRaw) {
      const step = normalizeStep(stepRaw);
      const stepSections = Array.from(document.querySelectorAll('[data-step]'));
      const backBtn = document.querySelector('[data-back]');

      const exists = stepSections.some(s => normalizeStep(s.getAttribute('data-step')) === step);
      const safeStep = exists ? step : 'home';

      stepSections.forEach(s => {
        s.classList.toggle('hidden', normalizeStep(s.getAttribute('data-step')) !== safeStep);
      });

      if (backBtn) backBtn.classList.toggle('hidden', safeStep === 'home');
    }

    // Init
    (function init() {
      const lang = detectLang();
      applyI18n(lang);
      showStep('home');

      // Event delegation
      document.addEventListener('click', (e) => {
        const closeEl = e.target.closest('[data-close-funnel]');
        if (closeEl) {
          e.preventDefault();
          // Ferme l'overlay côté WIX via postMessage (si intégré), sinon essaye window.close
          try { window.parent.postMessage({ type: 'NM_FUNNEL_CLOSE' }, '*'); } catch (_) {}
          return;
        }

        const backEl = e.target.closest('[data-back]');
        if (backEl) { e.preventDefault(); showStep('home'); return; }

        const gotoEl = e.target.closest('[data-goto]');
        if (gotoEl) { e.preventDefault(); showStep(gotoEl.getAttribute('data-goto')); }
      });

      document.addEventListener('keydown', (e) => {
        if (e.key === 'Escape') {
          try { window.parent.postMessage({ type: 'NM_FUNNEL_CLOSE' }, '*'); } catch (_) {}
        }
      });
    })();
